\section{Тест производительности}

	\begin{tabular}{| c | c |}

		1 4 1 & 7 \\
		2 5 3  & \\
		3 5 3 & \\
		4 5 10 & \\
		\hline
		4 4 & \\
		1 2 4 & \\
		1 3 3 & 4\\
		2 3 1 & \\
		3 4 5 & \\
		\hline
	\end{tabular}

\end{center}
